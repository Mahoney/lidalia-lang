25  package uk.org.lidalia.lang;
26  
27  import static com.google.common.base.Preconditions.checkNotNull;
28  
29  /**
30   * Static utility functions around Exception management.
31   */
32  public final class Exceptions {
33  
34      /**
35       * Because this method throws an unchecked exception, when it is called in a method with a return type the compiler
36       * does not know the method is exiting, requiring a further line to return null or throw an unchecked exception
37       * directly. This generified method allows this to be avoided by tricking the compiler by adding a return statement
38       * as so:
39       * <pre>
40       *     String someMethod() {
41       *         try {
42       *             somethingThatThrowsException();
43       *         } catch (Exception e) {
44       *             return throwUnchecked(e, null); // does not actually return, throws the exception
45       *         }
46       *     }
47       * </pre>
48       * @param toThrow The throwable that will be thrown, unwrapped and unchecked; must not be null
49       * @param returnType trick to persuade the compiler that a method returns appropriately - always pass null here
50       * @return Never returns, always throws the passed in exception
51       * @throws NullPointerException if toThrow is null
52       */
53      public static <T> T throwUnchecked(final Throwable toThrow, final T returnType) {
54          Exceptions.<RuntimeException>doThrowUnchecked(checkNotNull(toThrow));
55          throw new AssertionError("This code should be unreachable. Something went terribly wrong here!");
56      }
57  
58      /**
59       * @param toThrow The throwable that will be thrown, unwrapped and unchecked; must not be null
60       * @throws NullPointerException if toThrow is null
61       */
62      public static void throwUnchecked(final Throwable toThrow) {
63          throwUnchecked(toThrow, null);
64      }
65  
66      @SuppressWarnings("unchecked")
67      private static <T extends Throwable> void doThrowUnchecked(final Throwable toThrow) throws T {
68          throw (T) toThrow;
69      }
70  
71      private Exceptions() {
72          throw new UnsupportedOperationException("Not instantiable");
73      }
74  }
