25  /**
26   * A class that provides implementations of {@link #equals(Object)}, {@link #hashCode()} and {@link #toString()} for its subtypes.
27   * <p>
28   * These implementations are based on annotating the fields of the subtypes with the {@link Identity} annotation.
29   */
30  public abstract class RichObject {
31  
32      private static final int PRIME = 37;
33      private static final int INITIAL_HASHCODE_VALUE = 17;
34  
35      private static final LoadingCache<Class<?>, FluentIterable<FieldFacade>> IDENTITY_FIELDS =
36              CacheBuilder.newBuilder().weakKeys().softValues().build(new IdentityFieldLoader());
37      private static final Joiner FIELD_JOINER = Joiner.on(",");
38      private static final Function<Object, Integer> toHashCode = new Function<Object, Integer>() {
39          @Override
40          public Integer apply(final Object fieldValue) {
41              return fieldValue.hashCode();
42          }
43      };

45      /**
46       * Applies equality rules on the following basis (in addition to the rules in {@link Object#equals(Object)}):
47       * <ul>
48       * <li> other's runtime class must be the same, a super or a sub type of the runtime class of this instance
49       * <li> other's runtime class must have exactly the same set of fields annotated with {@link Identity} as those on the runtime
50       *      class of this instance, where the set of fields in each case comprises those on the class and all of its superclasses
51       * <li> the value of any field annotated with {@link Identity} on this must be equal to the value of the same field on other
52       * </ul>
53       * <p>
54       * The practical result of this is that an instance of subtype B of subtype A of RichObject can only be equal to an instance
55       * of subtype A if B does not annotate any of its fields with {@link Identity}.
56       *
57       * @param other the object to compare against
58       * @return true if the other type is logically equal to this
59       */
60      @Override public final boolean equals(final Object other) {
61          // Usual equals checks
62          if (other == this) {
63              return true;
64          }
65          if (other == null) {
66              return false;
67          }
68  
69          // One of the two must be a subtype of the other
70          if (!(other instanceof RichObject) || !inSameClassHierarchy(getClass(), other.getClass())) {
71              return false;
72          }
73  
74          final RichObject that = (RichObject) other;
75  
76          // They must have precisely the same set of identity members to meet the
77          // symmetric & transitive requirement of equals
78          final FluentIterable<FieldFacade> fieldsOfThis = fields();
79          return fieldsOfThis.toSet().equals(that.fields().toSet())
80                  && fieldsOfThis.allMatch(hasEqualValueIn(that));
81      }
82  
83      private FluentIterable<FieldFacade> fields() {
84          try {
85              return IDENTITY_FIELDS.get(getClass());
86          } catch (ExecutionException e) {
87              return throwUnchecked(e.getCause(), null);
88          }
89      }
90  
91      private Predicate<FieldFacade> hasEqualValueIn(final RichObject other) {
92          return new Predicate<FieldFacade>() {
93              @Override
94              public boolean apply(final FieldFacade field) {
95                  return valueOf(field).equals(other.valueOf(field));
96              }
97          };
98      }

100     /**
101      * Default implementation of hashCode - can be overridden to provide more efficient ones provided the contract specified
102      * in {@link Object#hashCode()} is maintained with respect to {@link #equals(Object)}.
103      *
104      * @return hash code computed from the hashes of all the fields annotated with {@link Identity}
105      */
106     @Override public int hashCode() {
107         int result = INITIAL_HASHCODE_VALUE;
108         for (final FieldFacade field : fields()) {
109             final int toAdd = valueOf(field).transform(toHashCode).or(0);
110             result = PRIME * result + toAdd;
111         }
112         return result;
113     }

115     /**
116      * Default implementation of toString.
117      *
118      * @return a string in the form ClassName[field1=value1,field2=value2] where the fields are those annotated with
119      * {@link Identity}
120      */
121     @Override public String toString() {
122         final Iterable<String> fieldsAsStrings = fields().transform(toStringValueOfField());
123         return getClass().getSimpleName()+"["+FIELD_JOINER.join(fieldsAsStrings)+"]";
124     }

126     private Function<FieldFacade, String> toStringValueOfField() {
127         return new Function<FieldFacade, String>() {
128             @Override
129             public String apply(final FieldFacade field) {
130                 return field.getName() + "=" + valueOf(field).or("absent");
131             }
132         };
133     }

135     private Optional<Object> valueOf(final FieldFacade field) {
136         return field.valueOn(this);
137     }

139     private static class IdentityFieldLoader extends CacheLoader<Class<?>, FluentIterable<FieldFacade>> {
140 
141         @Override
142         public FluentIterable<FieldFacade> load(final Class<?> key) {
143             return FluentIterable.from(doLoad(key));
144         }
145 
146         private static final Predicate<FieldFacade> onlyIdentityFields = new Predicate<FieldFacade>() {
147             @Override
148             public boolean apply(final FieldFacade field) {
149                 return field.isIdentityField();
150             }
151         };
152 
153         private static final Function<Field, FieldFacade> toFieldFacade = new Function<Field, FieldFacade>() {
154             @Override
155             public FieldFacade apply(final Field field) {
156                 return new FieldFacade(field);
157             }
158         };
159 
160         private static final Function<Class<?>, Set<FieldFacade>> toFieldSet = new Function<Class<?>, Set<FieldFacade>>() {
161             @Override
162             public Set<FieldFacade> apply(final Class<?> input) {
163                 return doLoad(input);
164             }
165         };
166 
167         private static Set<FieldFacade> doLoad(final Class<?> key) {
168             final ImmutableSet<FieldFacade> localIdentityFieldSet = FluentIterable.from(asList(key.getDeclaredFields()))
169                     .transform(toFieldFacade)
170                     .filter(onlyIdentityFields)
171                     .toSet();
172             final Optional<? extends Class<?>> superClass = fromNullable(key.getSuperclass());
173             final Set<FieldFacade> superIdentityFieldSet = superClass.transform(toFieldSet).or(ImmutableSet.<FieldFacade>of());
174             return Sets.union(localIdentityFieldSet, superIdentityFieldSet);
175         }
176     }

178     private static class FieldFacade extends WrappedValue {
179         private final Field field;
180 
181         FieldFacade(final Field field) {
182             super(field);
183             this.field = field;
184         }
185 
186         public Optional<Object> valueOn(final Object target) {
187             try {
188                 if (!field.isAccessible()) {
189                     makeAccessible();
190                 }
191                 return fromNullable(field.get(target));
192             } catch (IllegalAccessException e) {
193                 throw new IllegalStateException(field+" was not accessible; all fields should be accessible", e);
194             }
195         }
196 
197         public String getName() {
198             return field.getName();
199         }
200 
201         public boolean isIdentityField() {
202             return field.isAnnotationPresent(Identity.class);
203         }
204 
205         private void makeAccessible() {
206             doPrivileged(new PrivilegedAction<Void>() {
207                 @Override
208                 public Void run() {
209                     field.setAccessible(true);
210                     return null;
211                 }
212             });
213         }
214     }
