9    * Wraps and exposes the lazily evaluated result of an expression passed as a Callable to the constructor.
10   *
11   * @param <T> the type of the value wrapped by this class
12   */
13  public class LazyValue<T> implements Callable<T> {
14  
15      private final FutureTask<T> supplier;
16  
17      /**
18       * @param supplier contains the expression (as a Callable) that will create the result; this will be evaluated once
19       *                 and only once
20       */
21      public LazyValue(final Callable<T> supplier) {
22          this.supplier = new FutureTask<>(supplier);
23      }
24  
25      /**
26       * @throws Exception thrown by the original caller, unwrapped
27       * @return the result of the expression passed to the constructor; this is evaluated the first time this method is called,
28       *         subsequent calls return the same result without re-evaluating it. Concurrent calls block until the first execution
29       *         is complete.
30       */
31      @Override
32      public T call() {
33          supplier.run();
34          try {
35              return supplier.get();
36          } catch (Exception e) {
37              return throwUnchecked(e.getCause(), null);
38          }
